
create database DBEmpleados

use DBEmpleados

create table Area(
IdArea int primary key identity,
NombreArea varchar(50),
FechaCreacion datetime default getdate()
)


create table Empleados(
IdEmpleado int primary key identity,
NombreEmpleado varchar(50),
ApellidoEmpleado varchar(50),
TipoDocumento varchar,
NumeroDocumento varchar(50),
IdArea int references Area(IdArea),
FechaContrato datetime,
FechaCreacionEmpleado datetime default getdate()
)
